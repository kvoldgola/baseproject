\renewenvironment{ticket}{
    \par
    \parindent=0mm
    \penalty -9999
    \vskip 6mm
    \begin{minipage}{\textwidth}
        \begin{multicols}{\value{colsnumber}}
            \fbox{%
                \kern 2mm%
                {\Huge\bfseries\sffamily\arabic{tick}}%
                \kern 2mm%
                \small
                \sffamily
                \begin{tabular}[b]{ll}
                    \firstline
                    \\
                    \secondline
                \end{tabular}%
            }
            \hbadness=10000
            \interlinepenalty=10000
            \par
}{
        \end{multicols}
        \vskip -1mm
        \hbox to \textwidth {\leaders \hbox{\hskip 2.5mm \rule{1.5mm}{0.1mm}} \hfil}
    \end{minipage}
    \par
}
